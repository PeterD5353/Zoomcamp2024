SELECT
    gtd.*,
    pickup."LocationID" AS PULocationID,
    pickup."Borough" AS PUBorough,
    pickup."Zone" AS PUZone,
    dropoff."LocationID" AS DOLocationID,
    dropoff."Borough" AS DOBorough,
    dropoff."Zone" AS DOZone
FROM
    green_taxi_data gtd
JOIN
    zones pickup ON gtd."PULocationID" = pickup."LocationID"
JOIN
    zones dropoff ON gtd."DOLocationID" = dropoff."LocationID";
